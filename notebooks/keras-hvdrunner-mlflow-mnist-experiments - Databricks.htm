# MNIST CNN with Horovod Hooks
def runCNN_hvd():
  # Keras Initializers 
  keras.initializers.Initializer()

  # Horovod: initialize Horovod.
  hvd.init()

  # Horovod: pin GPU to be used to process local rank (one GPU per process)
  config = tf.ConfigProto()
  config.gpu_options.allow_growth = True
  config.gpu_options.visible_device_list = str(hvd.local_rank())
  
  # Initialize TensorFlow Session
  sess = tf.Session(config=config)
  K.set_session(sess)
  
  # Horovod: adjust number of epochs based on number of GPUs.
  epochs = int(math.ceil(12.0 / hvd.size()))
  print('epochs: ', epochs)

  # Including MLflow
  import mlflow
  import mlflow.keras
  import os

  # Configure Databricks MLflow environment
  #    This is my (denny.lee) personal token so you will want to generate yours
  mlflow.set_tracking_uri("databricks")
  os.environ['DATABRICKS_HOST'] = databricks_host
  os.environ['DATABRICKS_TOKEN'] = databricks_token
    
  # Building up our CNN
  model = Sequential()
  
  # Convolution Layer
  model.add(Conv2D(32, kernel_size=(3, 3),
                 activation='relu',
                 input_shape=input_shape)) 
  
  # Convolution layer
  model.add(Conv2D(64, (3, 3), activation='relu'))
  
  # Pooling with stride (2, 2)
  model.add(MaxPooling2D(pool_size=(2, 2)))
  
  # Delete neuron randomly while training (remain 75%)
  #   Regularization technique to avoid overfitting
  model.add(Dropout(0.25))
  
  # Flatten layer 
  model.add(Flatten())
  
  # Fully connected Layer
  model.add(Dense(128, activation='relu'))
  
  # Delete neuron randomly while training (remain 50%) 
  #   Regularization technique to avoid overfitting
  model.add(Dropout(0.5))
  
  # Apply Softmax
  model.add(Dense(num_classes, activation='softmax'))

  # Choose Optimizer + Adjusting Learning Rate based on Number of GPUs ala Horovod
  if choose_optimizer == 5:
      optimizer = keras.optimizers.Adadelta(lr=lr*hvd.size(), rho=0.95, epsilon=None, decay=0.0)
      optimizerdesc = "adadelta"
  elif choose_optimizer == 1:
      optimizer = keras.optimizers.SGD(lr=lr*hvd.size(), momentum=0.0, decay=0.0, nesterov=False)
      optimizerdesc = "sgd"
  elif choose_optimizer == 2:
      optimizer = keras.optimizers.SGD(lr=lr*hvd.size(), momentum=0.9, decay=0.0, nesterov=True)
      optimizerdesc = "nag"
  elif choose_optimizer == 3:
      optimizer = keras.optimizers.RMSprop(lr=lr*hvd.size(), rho=0.95, epsilon=None, decay=0.0)
      optimizerdesc = "rmsprop"
  elif choose_optimizer == 4:
      optimizer = keras.optimizers.Adam(lr=lr*hvd.size(), beta_1=0.9, beta_2=0.999, epsilon=None, decay=0.0, amsgrad=False)
      optimizerdesc = "adam"

  # Add Horovod Distributed Optimizer.
  optimizer = hvd.DistributedOptimizer(optimizer)

  # Record Start time
  start_time = time.time()

  # Loss function (crossentropy) and Optimizer (Adadelta)
  model.compile(loss=keras.losses.categorical_crossentropy,
              optimizer=optimizer,
              metrics=['accuracy'])

  # Only perform these tasks on worker 0
  if hvd.rank() == 0:
    # Configure for TensorBoard visualization
    #   Reference: [Monitor progress of your Keras based neural network using Tensorboard](https://bit.ly/2C36EBJ)
    tensorboard = TensorBoard(log_dir=output_dir)

    callbacks = [
      # Horovod: broadcast initial variable states from rank 0 to all other processes.
      # This is necessary to ensure consistent initialization of all workers when
      # training is started with random weights or restored from a checkpoint.
      hvd.callbacks.BroadcastGlobalVariablesCallback(0),
      tensorboard,
    ]
    
    # Horovod: save checkpoints only on worker 0 to prevent other workers from corrupting them.
    callbacks.append(keras.callbacks.ModelCheckpoint('./checkpoint-{epoch}.h5'))
    
  else:   
    callbacks = [
      # Horovod: broadcast initial variable states from rank 0 to all other processes.
      # This is necessary to ensure consistent initialization of all workers when
      # training is started with random weights or restored from a checkpoint.
      hvd.callbacks.BroadcastGlobalVariablesCallback(0),
    ]
  
  # Fit our model
  model.fit(x_train, y_train,
          batch_size=batch_size,
          callbacks=callbacks,
          epochs=epochs,
          verbose=1,
          validation_data=(x_test, y_test))

  # Evaluate our model
  score = model.evaluate(x_test, y_test, verbose=0)

  # Record Complete Time and determine Elapsed Time
  complete_time = time.time()
  elapsed_time = complete_time - start_time

  # MLflow Tracking (Log only from Worker 0)
  if hvd.rank() == 0:    
    # Write out TensorBoard events
    # writer = tf.summary.FileWriter(log_dir, graph=sess.graph)
    
    # Log events to MLflow
    with mlflow.start_run(run_uuid = active_run_uuid) as run:
      # Log MLflow Parameters
      mlflow.log_param("mode", process_mode)
      mlflow.log_param("epochs", epochs)
      mlflow.log_param("batch_size", batch_size)
      mlflow.log_param("num_classes", num_classes)
      mlflow.log_param("lr", lr)
      mlflow.log_param("optimizer_id", choose_optimizer)
      mlflow.log_param("optimizer_desc", optimizerdesc)

      # Log MLflow Metrics
      mlflow.log_metric("Test Loss", score[0])
      mlflow.log_metric("Test Accuracy", score[1])
      mlflow.log_metric("Duration", elapsed_time)
      
      # Log Model
      mlflow.keras.log_model(model, "models")
      
      # Log TF events to MLflow
      mlflow.log_artifacts(output_dir, artifact_path="tf.events")
