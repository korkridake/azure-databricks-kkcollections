  2. The first convolution layer maps one grayscale image to 32 feature maps using the activation function
  3. The second convolution layer maps the image to 64 feature maps using the activation function

def runCNN_arch(kernel_set = 3, pool_set = 2, flag_dropout1 = True, flag_bn1 = False, flag_dropout2 = True, flag_bn2 = False, verbose = 0):
  # Building up our CNN
  model = Sequential()
  
  # Convolution Layer
  if kernel_set == 5:
    model.add(Conv2D(32, kernel_size=(5, 5), input_shape=input_shape))
  else:
    model.add(Conv2D(32, kernel_size=(3, 3), input_shape=input_shape))
  
  # Use Batch Normalization (typically in concern with removing dropout)
  if flag_bn1 == True:
    model.add(BatchNormalization())
  
  # Activation Function
  model.add(Activation('relu'))

  # Convolution layer
  if kernel_set == 5:
    model.add(Conv2D(64, kernel_size=(5, 5)))
  else:
    model.add(Conv2D(64, kernel_size=(3, 3)))
  
  # Use Batch Normalization (typically in concern with removing dropout)
  if flag_bn2 == True:
    model.add(BatchNormalization())
  
  # Activation Function
  model.add(Activation('relu'))
  
  # Pooling with stride (2, 2)
  if pool_set == 4:
    model.add(MaxPooling2D(pool_size=(4, 4)))
  else: 
    model.add(MaxPooling2D(pool_size=(2, 2)))
  
  # Delete neuron randomly while training (remain 75%)
  #   Regularization technique to avoid overfitting
  if flag_dropout1 == True:
    model.add(Dropout(0.25))
  
  # Flatten layer 
  model.add(Flatten())
  
  # Fully connected Layer
  model.add(Dense(128, activation='relu'))
  
  # Delete neuron randomly while training (remain 50%) 
  #   Regularization technique to avoid overfitting
  if flag_dropout2 == True: 
    model.add(Dropout(0.5))
  
  # Apply Softmax
  model.add(Dense(num_classes, activation='softmax'))

  # Loss function (crossentropy) and Optimizer (Adadelta)
  model.compile(loss=keras.losses.categorical_crossentropy,
              optimizer=keras.optimizers.Adadelta(),
              metrics=['accuracy'])
  
  # Fit our model
  model.fit(x_train, y_train,
          batch_size=batch_size,
          epochs=epochs,
          verbose=verbose,
          validation_data=(x_test, y_test))

  # Evaluate our model
  score = model.evaluate(x_test, y_test, verbose=0)
    
  # Return
  return score
